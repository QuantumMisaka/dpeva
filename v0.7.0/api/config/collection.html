pydantic model dpeva.config.CollectionConfig[源代码]#

Configuration for Collection (Active Learning) Workflow.

field config_path: Path | None = None#
field desc_dir: Path [Required]#

Descriptor directory.

field direct_k: int = 1#
field direct_n_clusters: int | None = None#

Target number of clusters for Standard DIRECT.

field direct_thr_init: float = 0.5#
field fig_dpi: int = 300#

DPI for visualization figures.

field num_models: int = 4#

Number of models for UQ calculation.

field project: str = './'#

Project directory.

field results_prefix: str = 'results'#

Output file prefix.

field root_savedir: Path = PosixPath('dpeva_uq_result')#
field sampler_type: Literal['direct', '2-direct'] = 'direct'#
field step1_n_clusters: int | None = None#

Number of clusters for Step 1. If None, dynamic clustering based on threshold.

field step1_threshold: float = 0.5#

Threshold for Step 1 Birch clustering.

field step2_k: int = 5#

Number of samples per atomic cluster in Step 2.

field step2_n_clusters: int | None = None#

Number of clusters for Step 2. If None, dynamic clustering based on threshold.

field step2_selection: Literal['smallest', 'center', 'random'] = 'smallest'#

Selection criteria for Step 2. 'smallest': fewest atoms; 'center': closest to cluster centroid; 'random': random sampling.

field step2_threshold: float = 0.1#

Threshold for Step 2 Birch clustering.

field testdata_dir: Path [Required]#

Test data directory.

field testing_dir: str = 'test_results'#
field training_data_dir: Path | None = None#
field training_desc_dir: Path | None = None#
field uq_auto_bounds: Dict[str, Dict[str, float]] [Optional]#
field uq_qbc_trust_hi: float | None = None#
field uq_qbc_trust_lo: float | None = None#
field uq_qbc_trust_ratio: float | None = None#
field uq_qbc_trust_width: float | None = None#
field uq_rnd_rescaled_trust_hi: float | None = None#
field uq_rnd_rescaled_trust_lo: float | None = None#
field uq_rnd_rescaled_trust_ratio: float | None = None#
field uq_rnd_rescaled_trust_width: float | None = None#
field uq_select_scheme: Literal['tangent_lo', 'strict', 'circle_lo', 'crossline_lo', 'loose'] = 'tangent_lo'#

Strategy for selecting high-uncertainty data. Options: tangent_lo, strict, circle_lo, etc.

field uq_trust_mode: Literal['auto', 'manual', 'no_filter'] = 'auto'#
field uq_trust_ratio: float = 0.33#
field uq_trust_width: float = 0.25#
