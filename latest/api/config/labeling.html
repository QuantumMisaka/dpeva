Labeling Configuration#

pydantic model dpeva.config.LabelingConfig[源代码]#

Configuration for Labeling Workflow (FP Calculation).

field attempt_params: List[Dict[str, Any]] = [{}]#

Parameters for each attempt (index=0 is initial).

field cleaning_thresholds: Dict[str, float | None] = {'cohesive_energy': nan, 'force': 40.0, 'max_atoms': 255, 'stress': 40.0}#
field dft_params: Dict[str, Any] [Optional]#

ABACUS INPUT parameters.

field existing_training_data_path: Path | None = None#

Path to existing training dataset.

field input_data_path: Path [Required]#

Path to input dataset (dpdata compatible).

field integration_deduplicate: bool = False#

Whether to enable deduplication in integration.

field integration_enabled: bool = False#

Whether to integrate cleaned data into next-gen training data.

field integration_output_format: str = 'deepmd/npy/mixed'#

Output format for merged training dataset (deepmd/npy or deepmd/npy/mixed)

field kpt_criteria: int = 25#

K-point density parameter (K*L). Determines grid density: k_i = ceil(criteria / lattice_i).

field mag_map: Dict[str, float] [Optional]#

Initial magnetic moments per element (e.g. {'Fe': 5.0})

field merged_training_data_path: Path | None = None#

Output path for merged training dataset.

field orb_dir: str [Required]#

Directory containing Orb files.

field orb_map: Dict[str, str] [Optional]#

Dictionary mapping element symbols to orbital filenames (e.g., {'Fe': 'Fe.orb'}).

field output_format: str = 'deepmd/npy'#

Output format for cleaned/anomalies dataset (deepmd/npy or deepmd/npy/mixed)

field pp_dir: str [Required]#

Directory containing PP files.

field pp_map: Dict[str, str] [Optional]#

Dictionary mapping element symbols to pseudopotential filenames (e.g., {'Fe': 'Fe.upf'}).

field ref_energies: Dict[str, float] [Optional]#

Reference energies per element.

field tasks_per_job: int = 50#

Number of tasks per packed job.

field vacuum_thickness: float = 6.3#

Minimum vacuum thickness in Angstroms for surface/cluster models.

field work_dir: Path [Required]#

Path to labeling working directory.
